ptDelete

Note:

ptDelete is a directive that you can use this to delete record to database simply...


Arguments:

Param Type Details
element stringDOMElement

HTML string or DOMElement to be wrapped into jQuery.

Usage:

The normalization process is as follows:

Controller:


            
var testController = angular.module('testController',[])
                testController.controller('insertController',function($scope){
                $scope.insertData = {
                'id':'1',
                'name': 'insert'
                }
                $scope.insertRedirect={
                'url':'#',
                'param':'1'
                }
                })

Service:


            
testService.factory('insertService',function($resource){
                return $resource("#");
                })

Html:


            


                <div ng-controller="insertController">
                <button pt-insert data="insertData" service="insertService" flag="0" redirect="insertRedirect">add
                </button>
                </div>
